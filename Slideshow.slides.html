def count_distinct_countries(str):
    """
    We will count 1 2 and 3 in which:
    1 means all father mother and child were born in the same country
    2 means two out of the three were born in different countries
    3 means all of them were originally from 3 different countries
    """
    s = set()
    for t in str.split(","):
        s.add(t)
    return len(s)

# The three country names of mother, father, and the student are merged to one string for the later process
df_clean['Merged Birth Countries'] = df_clean['Birth Country of Mother'] + "," +  df_clean['Birth Country of Father'] + "," + df_clean['Birth Country'];

# we apply the previously defined function to the merged names of countries to compute the representing number of 
# different birth countries 
df_clean['Family country combination'] = df_clean['Merged Birth Countries'].apply(count_distinct_countries);
# Convert values to descriptive text
df_clean['Family country combination'] = df_clean['Family country combination'].replace({1: 'Same Country', 2: 'Two Countries', 3:'Different Countries'})

res = df_clean.groupby('Family country combination').sum()['Birth Year'].sort_values()
res.plot.bar(logy=True)
plt.title('Family country combination')
